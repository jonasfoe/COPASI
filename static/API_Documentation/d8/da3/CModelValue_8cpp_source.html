143 bool CModelEntity::compile()
144 {
145  bool success = true;
146 
147  std::set< const CCopasiObject * > NoDependencies;
148  std::vector< CCopasiContainer * > listOfContainer;
149  listOfContainer.push_back(mpModel);
150  CCopasiDataModel* pDataModel = NULL;
151 
152  switch (mStatus)
153  {
154  case ASSIGNMENT:
155  success &= mpExpression->compile(listOfContainer);
156  mpValueReference->setDirectDependencies(mpExpression->getDirectDependencies());
157 
158  pdelete(mpInitialExpression);
159  pDataModel = getObjectDataModel();
160  assert(pDataModel != NULL);
161  mpInitialExpression = CExpression::createInitialExpression(*mpExpression, pDataModel);
162  mpInitialExpression->setObjectName("InitialExpression");
163  add(mpInitialExpression, true);
164  break;
165 
166  case ODE:
167  mpValueReference->addDirectDependency(this);
168 
169  success &= mpExpression->compile(listOfContainer);
170  mpRateReference->setDirectDependencies(mpExpression->getDirectDependencies());
171  break;
172 
173  default:
174  break;
175  }
176 
177  // Here we handle initial expressions for all types.
178  if (mpInitialExpression != NULL &&
179  mpInitialExpression->getInfix() != "")
180  {
181  success &= mpInitialExpression->compile(listOfContainer);
182  mpIValueReference->setDirectDependencies(mpInitialExpression->getDirectDependencies());
183 
184  // If we have a valid initial expression, we update the initial value.
185  // In case the expression is constant this suffices other are updated lated again.
186  if (mpInitialExpression->isUsable())
187  *mpIValue = mpInitialExpression->calcValue();
188  }
189  else
190  {
191  mpIValueReference->setDirectDependencies(NoDependencies);
192  }
193 
194  return success;
195 }

395 void CModelEntity::setStatus(const CModelEntity::Status & status)
396 {
397  if (mStatus != status)
398  {
399  if (mpModel != NULL)
400  mpModel->setCompileFlag(true);
401 
402  // An assignment may not have an initial expression.
403  // However, internally we always create one, which need
404  // to be deleted.
405  if (mStatus == ASSIGNMENT)
406  pdelete(mpInitialExpression);
407 
408  mStatus = status;
409  this->setValuePtr(mpValue);
410 
411  if (mpModel != NULL)
412  mpModel->setCompileFlag(true);
413 
414  std::set< const CCopasiObject * > NoDependencies;
415 
416  setDirectDependencies(NoDependencies);
417  clearRefresh();
418 
419  mpIValueReference->setDirectDependencies(NoDependencies);
420 
421  mpValueReference->setDirectDependencies(NoDependencies);
422  mpValueReference->clearRefresh();
423 
424  mpRateReference->setDirectDependencies(NoDependencies);
425  mpRateReference->clearRefresh();
426  CCopasiDataModel* pDataModel = NULL;
427 
428  switch (mStatus)
429  {
430  case ASSIGNMENT:
431 
432  if (mpExpression == NULL)
433  mpExpression = new CExpression("Expression", this);
434 
435  pdelete(mpInitialExpression);
436  pDataModel = getObjectDataModel();
437  mpInitialExpression = CExpression::createInitialExpression(*mpExpression, pDataModel);
438  mpInitialExpression->setObjectName("InitialExpression");
439  add(mpInitialExpression, true);
440 
441  mpValueReference->setDirectDependencies(mpExpression->getDirectDependencies());
442  mpValueReference->setRefresh(this, &CModelEntity::calculate);
443 
444  mRate = std::numeric_limits<C_FLOAT64>::quiet_NaN();
445 
446  mUsed = true;
447  break;
448 
449  case ODE:
450 
451  if (mpExpression == NULL)
452  mpExpression = new CExpression("Expression", this);
453 
454  mpRateReference->setDirectDependencies(mpExpression->getDirectDependencies());
455  mpRateReference->setRefresh(this, &CModelEntity::calculate);
456 
457  mUsed = true;
458  break;
459 
460  case REACTIONS:
461  pdelete(mpExpression);
462 
463  mUsed = true;
464  break;
465 
466  case TIME:
467  pdelete(mpExpression);
468 
469  mUsed = true;
470  break;
471 
472  case FIXED:
473  pdelete(mpExpression);
474 
475  mRate = 0.0;
476 
477  mUsed = false;
478  break;
479  }
480  }
