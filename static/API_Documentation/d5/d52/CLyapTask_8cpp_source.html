84 void CLyapTask::initObjects()
85 {
86  size_t i;
87 
88  mExponents.resize(LYAP_NUM_REF);
89 
90  for (i = 0; i < LYAP_NUM_REF; ++i) mExponents[i] = 0.0;
91 
92  mvExpRef.resize(LYAP_NUM_REF);
93 
94  mLocalExponents.resize(LYAP_NUM_REF);
95 
96  for (i = 0; i < LYAP_NUM_REF; ++i) mLocalExponents[i] = 0.0;
97 
98  mvLocExpRef.resize(LYAP_NUM_REF);
99 
100  addVectorReference("Exponents", mExponents, CCopasiObject::ValueDbl);
101  addVectorReference("Local exponents", mLocalExponents, CCopasiObject::ValueDbl);
102 
103  for (i = 0; i < LYAP_NUM_REF; ++i)
104  {
105  std::ostringstream sss;
106  sss << "Exponent " << i + 1;
107  mvExpRef[i] = dynamic_cast<CCopasiObjectReference<C_FLOAT64> *>
108  (addObjectReference(sss.str(), mExponents.array()[i], CCopasiObject::ValueDbl));
109  assert(mvExpRef[i]);
110 
111  std::ostringstream sss2;
112  sss2 << "Local exponent " << i + 1;
113  mvLocExpRef[i] = dynamic_cast<CCopasiObjectReference<C_FLOAT64> *>
114  (addObjectReference(sss2.str(), mLocalExponents.array()[i], CCopasiObject::ValueDbl));
115  assert(mvLocExpRef[i]);
116  }
117 
118  addObjectReference("Sum of exponents", mSumOfExponents, CCopasiObject::ValueDbl);
119  addObjectReference("Sum of local exponents", mSumOfLocalExponents, CCopasiObject::ValueDbl);
120  //addObjectReference("Local divergence", mDivergence, CCopasiObject::ValueDbl);
121  addObjectReference("Interval divergence", mIntervalDivergence, CCopasiObject::ValueDbl);
122  addObjectReference("Average divergence", mAverageDivergence, CCopasiObject::ValueDbl);

168 bool CLyapTask::process(const bool & useInitialValues)
169 {
170  if (useInitialValues)
171  mpLyapProblem->getModel()->applyInitialValues();
172 
173  output(COutputInterface::BEFORE);
174 
175  // bool flagProceed = true;
176  mPercentage = 0;
177 
178  if (mpCallBack)
179  {
180  mpCallBack->setName("performing lyapunov exponent calculation...");
181  C_FLOAT64 hundred = 100;
182  mhProcess = mpCallBack->addItem("Completion",
183  mPercentage,
184  &hundred);
185  }
186 
187  try
188  {
189  mpLyapMethod->calculate();
190  }
191  catch (CCopasiException & Exception)
192  {
193  //mpLyapProblem->getModel()->setState(*mpCurrentState);
194  mpLyapProblem->getModel()->updateSimulatedValues(true);
195 
196  calculationsBeforeOutput();
197  output(COutputInterface::DURING);
198 
199  if (mpCallBack) mpCallBack->finishItem(mhProcess);
200 
201  output(COutputInterface::AFTER);
202 
203  throw CCopasiException(Exception.getMessage());
204  }
205 
206  if (mpCallBack) mpCallBack->finishItem(mhProcess);
207 
208  calculationsBeforeOutput();
209  output(COutputInterface::AFTER);
210 
211  mResultAvailable = true;
212  mResultHasDivergence = mpLyapProblem->divergenceRequested();
213  mModelVariablesInResult = mpLyapProblem->getModel()->getState().getNumIndependent();
214  mNumExponentsCalculated = mpLyapProblem->getExponentNumber();
215 
216  return true;
217 }
