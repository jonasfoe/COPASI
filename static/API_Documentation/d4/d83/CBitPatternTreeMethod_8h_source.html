57  */
58  static inline void GCD(C_INT64 & m, C_INT64 & n)
59  {
60  assert(m > 0 && n > 0);
61 
62  while (m != n)
63  {
64  if (m > n)
65  {
66  m %= n;
67 
68  if (m == 0)
69  {
70  m = n;
71  }
72  }
73  else
74  {

76

78  {
79  n = m;
80  }
81  }
82  }
83  }
