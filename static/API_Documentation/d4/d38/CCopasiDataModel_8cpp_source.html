457  }
458 
459  if (autoSave || !success)
460  {
461  try
462  {
463  if (!XML.CCopasiXMLInterface::save(FileName, FileName))
464  return false;
465  }
466 
467  catch (...)
468  {
469  return false;
470  }
471  }
472 
473  if (!autoSave)
474  {
475  changed(false);
476  mData.mSaveFileName = CDirEntry::normalize(FileName);
477  }
478 
479  return true;

1902  containerIndex++)
1903  {
