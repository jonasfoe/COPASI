52  * Calculate the greatest common divisor (GCD) of 2 positive integers. On return
53  * m and n contain the GCD
54  * @param C_INT64 & m
55  * @param C_INT64 & n
56  */
57  static inline void GCD(C_INT64 & m, C_INT64 & n)
58  {
59  assert(m > 0 && n > 0);
60 
61  while (m != n)
62  {
63  if (m > n)
64  {
65  m %= n;
66 
67  if (m == 0)
68  {
69  m = n;
70  }
71  }
72  else
73  {
74  n %= m;
75 
76  if (n == 0)
77  {
78  n = m;
79  }
80  }
81  }
82  }
