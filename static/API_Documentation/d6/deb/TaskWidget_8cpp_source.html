395 bool TaskWidget::commonRunTask()
396 {
397  // Initialize the task
398  try
399  {
400  assert(CCopasiRootContainer::getDatamodelList()->size() > 0);
401 
402  if (!mpTask->initialize(CCopasiTask::OUTPUT_UI, (*CCopasiRootContainer::getDatamodelList())[0], NULL))
403  throw CCopasiException(CCopasiMessage::peekLastMessage());
404  }
405 
406  catch (CCopasiException & /*Exception*/)
407  {
408  if (CCopasiMessage::peekLastMessage().getNumber() != MCCopasiMessage + 1)
409  {
410  if (mProgressBar != NULL) mProgressBar->finish();
411 
412  CQMessageBox::critical(this, "Initialization Error",
413  CCopasiMessage::getAllMessageText().c_str(),
414  QMessageBox::Ok | QMessageBox::Default, QMessageBox::NoButton);
415 
416  finishTask();
417  return false;
418  }
419  }
420 
421  if (CCopasiMessage::getHighestSeverity() > CCopasiMessage::ERROR)
422  {
423  if (mProgressBar != NULL) mProgressBar->finish();
424 
425  CQMessageBox::critical(this, "Initialization Error",
426  CCopasiMessage::getAllMessageText().c_str(),
427  QMessageBox::Ok | QMessageBox::Default, QMessageBox::NoButton);
428 
429  finishTask();
430  return false;
431  }
432 
433  if (CCopasiMessage::getHighestSeverity() > CCopasiMessage::COMMANDLINE)
434  {
435  C_INT Result =
436  CQMessageBox::question(this, "Initialization Warning",
437  CCopasiMessage::getAllMessageText().c_str(),
438  QMessageBox::Ignore | QMessageBox::Abort, QMessageBox::Ignore);
439 
440  if (Result == QMessageBox::Abort)
441  {
442  finishTask();
443  return false;
444  }
445  }
446 
447  CCopasiMessage::clearDeque();
448 
449  // Execute the task
450  mpTaskThread->start();
451 
452  return true;
453 }
