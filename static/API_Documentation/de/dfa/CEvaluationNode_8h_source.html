344  /**
345  * Structure holding the precedence information
346  * The Precedence Table
347  * ====================
348  *
349  * Nodes of type WHITESPACE are ignored.
350  * Higher right than left precedence means evaluation
351  * proceeds form left to right. Higher left precedence
352  * as can be found for the operator '^' means evaluation
353  * proceeds from right to left.
354  *
355  * left precedence | | right precedence
356  * -----------------+--------------+------------------
357  * | LOGIC: |
358  * 3 | .not. | 2
359  * 4 | .or. | 5
360  * 6 | .xor. | 7
361  * 8 | .and. | 9
362  * 10 | .eq. | 11
363  * 12 | .ne. | 13
364  * 14 | .gt. | 15
365  * 16 | .ge. | 17
366  * 18 | .lt. | 19
367  * 20 | .le. | 21
368  * -----------------+--------------+------------------
369  * | OPERATOR: |
370  * 22 | - | 23
371  * 22 | + | 23
372  * 26 | % | 27
373  * 28 | / | 29
374  * 28 | * | 29
375  * 33 | ^ | 32
376  * -----------------+--------------+------------------
377  * 35 | FUNCTION | 34
378  * 35 | MV_FUNCTION | 34
379  * 35 | CALL | 34
380  * 35 | CHOICE | 34
381  * -----------------+--------------+------------------
382  * 36 | NUMBER | 37
383  * 36 | CONSTANT | 37
384  * 36 | OBJECT | 37
385  * 36 | VARIABLE | 37
386  * -----------------+--------------+------------------
387  * | STRUCTURE: |
388  * ULONG_MAX | (| 1
389  * 0 | , | 1
390  * 0 |) | ULONG_MAX
391  */
