290 void CScanItemRandom::step()
291 {
292  C_FLOAT64 Value;
293 
294  //the index
295  if (mIndex > mNumSteps)
296  mFlagFinished = true;
297  else
298  {
299  C_FLOAT64 tmpF;
300 
301  switch (mRandomType)
302  {
303  case 0: // uniform
304  Value = mMin + mRg->getRandomCC() * mFaktor;
305 
306  if (mLog)
307  Value = exp(Value);
308 
309  break;
310 
311  case 1: // normal
312  tmpF = mRg->getRandomNormal01();
313  Value = mMin + tmpF * mMax;
314 
315  if (mLog)
316  Value = exp(Value);
317 
318  break;
319 
320  case 2: // poisson
321 
322  if (mMin < 0)
323  CCopasiMessage(CCopasiMessage::WARNING, "Invalid ScanItem: Requested Poisson random variable for negative argument: %lf", mMin);
324 
325  Value = mRg->getRandomPoisson(mMin);
326 
327  break;
328 
329  case 3: // gamma
330  Value = mRg->getRandomGamma(mMin, mMax);
331 
332  if (mLog)
333  Value = exp(Value);
334 
335  break;
336  }
337  }
338 
339  if (mpObject) mpObject->setObjectValue(Value);
340 
341  if (mpInitialObject) mpInitialObject->setObjectValue(Value);
342 
343  ++mIndex;
344 }

389 
390  if (mpTask == NULL) return false;
391 
392  cleanupScanItems();

398  std::set< const CCopasiObject * > TransientObjectSet;
399 
400  size_t i, imax = mpProblem->getNumberOfScanItems();
401  mContinueFromCurrentState = mpProblem->getContinueFromCurrentState();
402 
403  for (i = 0; i < imax; ++i)
404  {
405  CScanItem * pItem = CScanItem::createScanItemFromParameterGroup(mpProblem->getScanItem(i),
406  mpRandomGenerator,
407  mContinueFromCurrentState);
408 
409  if (pItem == NULL)
410  {
411  continue;
412  }
413 
414  mScanItems.push_back(pItem);
415  mTotalSteps *= pItem->getNumSteps() + 1;
416 
417  if (pItem->getObject() != NULL)
418  {
419  InitialObjectSet.insert(pItem->getObject());
420  }
421  }
422 
423  if (mContinueFromCurrentState)
424  {
425  mInitialRefreshes = CCopasiObject::buildUpdateSequence(mpProblem->getModel()->getUptoDateObjects(), InitialObjectSet);
426  }
427  else
428  {
429  mInitialRefreshes = mpProblem->getModel()->buildInitialRefreshSequence(InitialObjectSet);
430  }
431 
432  //set mLastNestingItem
433  mLastNestingItem = C_INVALID_INDEX;
434 
435  if (imax != 0)
436  {
437  //search from the end
438  size_t j;
439 
440  for (j = mScanItems.size() - 1; j != C_INVALID_INDEX; --j)
441  {
442  if (mScanItems[j]->isNesting())
443  {
444  mLastNestingItem = j;
445  break;
446  }
447  }
448  }
449 
450  return true;
