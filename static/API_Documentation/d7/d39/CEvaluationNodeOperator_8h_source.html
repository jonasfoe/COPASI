76  /**
77  * Calculate the numerical result of the node. It is assumed that
78  * all child nodes are up to date.
79  */
80  virtual inline void calculate()
81  {
82  switch (mType & 0x00FFFFFF)
83  {
84  case POWER:
85  mValue = pow(mpLeft->getValue(), mpRight->getValue());
86  break;
87 
88  case MULTIPLY:
89  mValue = mpLeft->getValue() * mpRight->getValue();
90  break;
91 
92  case DIVIDE:
93  mValue = mpLeft->getValue() / mpRight->getValue();
94  break;
95 
96  case MODULUS:
97 
98  if ((C_INT32) mpRight->getValue() == 0)
99  mValue = std::numeric_limits< C_FLOAT64 >::quiet_NaN();
100  else
101  mValue = (C_FLOAT64)(((C_INT32) mpLeft->getValue()) % ((C_INT32) mpRight->getValue()));
102 
103  break;
104 
105  case PLUS:
106  mValue = mpLeft->getValue() + mpRight->getValue();
107  break;
108 
109  case MINUS:
110  mValue = mpLeft->getValue() - mpRight->getValue();
111  break;
112 
113  default:
114  break;
115  }
116  }
