379 const C_FLOAT64 & COptMethodLevenbergMarquardt::evaluate()
380 {
381  // We do not need to check whether the parametric constraints are fulfilled
382  // since the parameters are created within the bounds.
383 
384  mContinue &= mpOptProblem->calculate();
385  mEvaluationValue = mpOptProblem->getCalculateValue();
386 
387  // when we leave either the parameter or functional domain
388  // we penalize the objective value by forcing it to be larger
389  // than the best value recorded so far.
390  if (mEvaluationValue < mBestValue &&
391  (!mpOptProblem->checkParametricConstraints() ||
392  !mpOptProblem->checkFunctionalConstraints()))
393  mEvaluationValue = mBestValue + mBestValue - mEvaluationValue;
394 
395  return mEvaluationValue;
396 }
