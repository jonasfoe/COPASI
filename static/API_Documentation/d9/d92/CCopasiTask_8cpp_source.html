207 bool CCopasiTask::initialize(const OutputFlag & of,
208  COutputHandler * pOutputHandler,
209  std::ostream * pOstream)
210 {
211  bool success = true;
212 
213  if (!mpProblem)
214  {
215  CCopasiMessage(CCopasiMessage::ERROR, MCCopasiTask + 1, getObjectName().c_str());
216  return false;
217  }
218 
219  if (!mpProblem->getModel())
220  {
221  CCopasiMessage(CCopasiMessage::ERROR, MCCopasiTask + 2, getObjectName().c_str());
222  return false;
223  }
224 
225  if (!mpMethod)
226  {
227  CCopasiMessage(CCopasiMessage::ERROR, MCCopasiTask + 3, getObjectName().c_str());
228  return false;
229  }
230 
231  if (!mpProblem->getModel()->compileIfNecessary(mpCallBack))
232  {
233  CCopasiMessage(CCopasiMessage::ERROR, MCCopasiTask + 4, mpProblem->getModel()->getObjectName().c_str());
234  return false;
235  }
236 
237  pdelete(mpInitialState);
238 
239  mpInitialState = new CState(mpProblem->getModel()->getInitialState());
240 
241  mDoOutput = of;
242  mpOutputHandler = pOutputHandler;
243 
244  if (mDoOutput == NO_OUTPUT ||
245  mpOutputHandler == NULL) return true;
246 
247  mOutputCounter = 0;
248 
249  if (mDoOutput & REPORT)
250  {
251  if (mReport.open(getObjectDataModel(), pOstream) &&
252  mReport.getTarget() != "")
253  mpOutputHandler->addInterface(&mReport);
254  else if (pOstream == NULL)
255  CCopasiMessage(CCopasiMessage::COMMANDLINE, MCCopasiTask + 5, getObjectName().c_str());
256  }
257 
258  std::vector< CCopasiContainer * > ListOfContainer;
259  ListOfContainer.push_back(this);
260 
261  CCopasiDataModel* pDataModel = getObjectDataModel();
262  assert(pDataModel != NULL);
263 
264  if (!mpOutputHandler->compile(ListOfContainer, pDataModel))
265  {
266  // Warning
267  CCopasiMessage(CCopasiMessage::WARNING, MCCopasiTask + 7);
268  success = false;
269  }
270 
271  return success;
272 }

331 void CCopasiTask::output(const COutputInterface::Activity & activity)
332 {
333  if (mpOutputHandler != NULL)
334  switch (activity)
335  {
336  case COutputInterface::DURING:
337 
338  if (mDoOutput != NO_OUTPUT)
339  {
340  mpOutputHandler->output(activity);
341  ++mOutputCounter;
342  }
343 
344  break;
345 
346  case COutputInterface::BEFORE:
347 
348  if (mDoOutput & OUTPUT_BEFORE)
349  mpOutputHandler->output(activity);
350 
351  break;
352 
353  case COutputInterface::AFTER:
354 
355  if (mDoOutput & OUTPUT_AFTER)
356  mpOutputHandler->output(activity);
357 
358  break;
359  }
